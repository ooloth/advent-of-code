Input = list[str]
Answer = int


def solution(input: Input) -> Answer:
    return len(input)


def test_solution() -> None:
    examples: list[tuple[Input, Answer]] = [
        (["a"], 1),
        (["a", "b"], 2),
        (["a", "b", "c"], 3),
    ]

    for input, expected in examples:
        assert solution(input) == expected
